package innerclass.demo.ch10_1;//: innerclass/Parcel2.java
// Returning a reference to an inner class.

public class Parcel2 {
  class Contents {
    private int i = 11;
    public int value() { return i; }
  }
  class Destination {
    private String label;
    Destination(String whereTo) {
      label = whereTo;
    }
    String readLabel() { return label; }
  }
  public Destination to(String s) {
    return new Destination(s);
  }
  public Contents contents() {
    return new Contents();
  }
  public void ship(String dest) {
    Contents c = contents();
    Destination d = to(dest);
    System.out.println(d.readLabel());
  }
  public static void main(String[] args) {
    Parcel2 p = new Parcel2();
    p.ship("Tasmania");
    Parcel2 q = new Parcel2();
    // Defining references to inner classes:
    Contents c = q.contents();
    System.out.println(c.value());
    Destination d = q.to("Borneo");
    System.out.println(d.readLabel());

    Parcel2_A a = new Parcel2_A();
    a.test();
  }
}

class Parcel2_A {
  public void test() {
    Parcel2 p = new Parcel2();
    p.ship("miaomiao");;
    Parcel2.Contents contents = p.contents();
    System.out.println(contents.value());
    Parcel2.Destination destination = p.to("test!");
    System.out.println(destination.readLabel());
  }
}
